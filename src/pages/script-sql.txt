SELECT mp.cod_medicamento_processo, proc.cod_processo AS cod_processo, pac.cod_paciente AS cod_paciente, med.cod_medicamento AS cod_medicamento, pac.nome_paciente AS nome_paciente, proc.numero_processo AS numero_processo, med.nome_medicamento AS nome_medicamento FROM medicamentos_processo mp JOIN processos proc ON mp.cod_processo = proc.cod_processo JOIN pacientes pac ON proc.cod_paciente = pac.cod_paciente JOIN medicamentos med ON proc.cod_medicamento = med.cod_medicamento;
